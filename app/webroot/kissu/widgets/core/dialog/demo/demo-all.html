7. 在指定位置弹出窗口


	var diag = new Dialog();
	diag.Width = 200;
	diag.Height = 100;
	diag.Modal = false;
	diag.Title = "在指定位置弹出窗口";
	diag.Top="100%";
	diag.Left="100%";
	diag.URL = "test.html";
	diag.show();
注：可使用数字或百分比（带百分比符号的字符串）来定义相对于当前窗口的位置，换算效果同css中用百分比定义背景图位置，如Left:"0%",Top:"0%"为左上，Left:"50%",Top:"50%"为居中，Left:"100%",Top:"100%"为右下。

9. 代替window.alert及window.confirm


	Dialog.alert("提示：你点击了一个按钮");

	Dialog.confirm('警告：您确认要XXOO吗？',function(){Dialog.alert("yeah，周末到了，正是好时候")});
注：Dialog.alert(msg, func, w, h)第二个参数为点击“确定”按钮后执行的函数。
Dialog.confirm(msg, funcOK, funcCal, w, h)第二个参数为点击“确定”按钮后执行的函数，第三个参数为点击“取消”按钮后执行的函数。

16. 动态改变窗口大小

		doc.close();
		diag.setSize(600,300);

	};
	diag.show();
	diag.okButton.value="改变窗口大小"

19. 窗体内的按钮操作父Dialog


	var diag = new Dialog();
	diag.Title = "窗体内的按钮操作父Dialog";
	diag.URL = "test.html";
	diag.show();
	var doc=diag.innerFrame.contentWindow.document;
	doc.open();
	doc.write('<html><body>\
		<input type="button" id="a" value="修改父Dialog尺寸" \
		onclick="ownerDialog.setSize(function(min,max){return Math.round(min+(Math.random()*(max-min)))}(300,800))" />\
		<input type="button" id="b" value="关闭父窗口" onclick="ownerDialog.close()" />\
		<input type="button" id="c" value="点击窗口取消按钮" onclick="ownerDialog.cancelButton.onclick()" />\
		</body></html>') ;
	doc.close();

20. 提交表单后，刷新父窗口


	var diag = new Dialog();
	diag.URL = "test.html";
	diag.OKEvent=function(){diag.innerDoc.getElementById('form1').submit();diag.submited=true;};
	diag.OnLoad=function(){
		if(diag.submited){
			alert('表单已经提交成功\n'+location+'\n接下来将刷新父窗口。');
			diag.openerWindow.location.reload();
            try{
				diag.close();
            }catch(e){}
		}
	};
	diag.show();
说明：如果要通过js来判断表单是否提交成功，可以在弹出框的页面内通过后台输出一个js变量如表单提交成功后在页面内输出<script>submitsuccess=true</script><div class="info success">表单提交成功</div> ，然后加上提交成功与否的判断，如上示例代码的第六行位置可改为
if(diag..innerWin.submitOK) alert('表单已经提交成功') else alert('表单提交失败')
